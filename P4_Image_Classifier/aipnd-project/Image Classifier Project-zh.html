def process_image(image):
    ''' Scales, crops, and normalizes a PIL image for a PyTorch model,
        returns an Numpy array
    '''
    # DONE: Process a PIL image for use in a PyTorch model
    #  thumbnail
    #   
    image.thumbnail((256, 256))
    # 224*224 crop image1
    #  https://pillow.readthedocs.io/en/latest/reference/Image.html#PIL.Image.Image.resize
    #  ,
    image1 = image.crop((16,16,240,240))
    # PIL image array 0-1  https://www.cnblogs.com/changbaishan/p/8432786.html
    np_image = np.array(image1) / 255
    #   
    np_image[:,:,0] = (np_image[:,:,0] - 0.485) / 0.229;
    np_image[:,:,1] = (np_image[:,:,1] - 0.456) / 0.224;
    np_image[:,:,2] = (np_image[:,:,2] - 0.406) / 0.225;
    #  http://www.cnblogs.com/sunshinewang/p/6893503.html
    #   0 1 ...
    # 224*224*3  3 0 
    pytorch_np_image = np_image.transpose((2,0,1))
    # np array...Tensor
    return pytorch_np_image

def imshow(image, ax=None, title=None):
    """Imshow for Tensor."""
    if ax is None:
        fig, ax = plt.subplots()
    
    # PyTorch tensors assume the color channel is the first dimension
    # but matplotlib assumes is the third dimension
    image = image.numpy().transpose((1, 2, 0))
    
    # Undo preprocessing
    mean = np.array([0.485, 0.456, 0.406])
    std = np.array([0.229, 0.224, 0.225])
    image = std * image + mean
    
    # Image needs to be clipped between 0 and 1 or it looks like noise when displayed
    image = np.clip(image, 0, 1)
    
    ax.imshow(image)
    
    return ax

def predict(image_path, model, topk=5):
    ''' Predict the class (or classes) of an image using a trained deep learning model.
    '''
    # DONE: Implement the code to predict the class from an image file
    model.eval() # 
    model.cpu() # GPU
    #  Tensor
    image = torch.FloatTensor(process_image(Image.open(image_path)))
    #   https://discuss.pytorch.org/t/expected-stride-to-be-a-single-integer-value-or-a-list-of-1-values-to-match-the-convolution-dimensions-but-got-stride-1-1/17140
    # 3*224*224 1*3*224*224
    # 
    a = Variable(image).cpu()
    a = a.unsqueeze(0)
    #  forward...cpu gpu cpu
    with torch.no_grad():
        output = new_model.forward(a)
        top_prob, top_labels = torch.topk(output, topk)
        # 
        top_prob = top_prob.exp()
    # 
    class_to_idx_inv = {new_model.class_to_idx[k]: k for k in new_model.class_to_idx}
    mapped_classes = list()
    # labels
    for label in top_labels.numpy()[0]:
        mapped_classes.append(class_to_idx_inv[label])
        
    return top_prob.numpy()[0], mapped_classes
